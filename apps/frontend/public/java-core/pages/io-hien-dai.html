Tác vụ I/O thường gặp trong Java hiện đại

Đọc và ghi file văn bản

// Đọc toàn bộ file
String content = Files.readString(Path.of("data.txt"), StandardCharsets.UTF_8);

// Đọc từng dòng
List<String> lines = Files.readAllLines(Path.of("data.txt"), StandardCharsets.UTF_8);

// Ghi file
Files.writeString(Path.of("output.txt"), "Hello World\n", 
    StandardCharsets.UTF_8, 
    StandardOpenOption.CREATE, 
    StandardOpenOption.APPEND);

Đọc dữ liệu từ web

// HTTP Client (Java 11+)
HttpClient client = HttpClient.newHttpClient();
HttpRequest request = HttpRequest.newBuilder()
    .uri(URI.create("https://api.example.com/data"))
    .build();

HttpResponse<String> response = client.send(request,
    HttpResponse.BodyHandlers.ofString());

IO.println(response.body());
